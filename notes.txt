1. Reverse a String with Built-In Functions 
--- JavaScript: 

    function reverseString(string) {
        let splitString = string.split(" ")
        let reverseArray = splitString.reverse()
        let joinArray = reverseArray.join(" ")
        return joinArray
    }
    reverseString('hello')

    -------or----------

    function reverseString(string) {
        return string.split(" ").reverse().join(" ");
    }
    reverseString("hello")
